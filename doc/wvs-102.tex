\section{$\alpha(s)$}

For most species $k$ we think of $\alpha$ as

\begin{equation}
\alpha(s) = \sum_k f^k(s)\, \beta^k(s) \text{ where }
 f^k(s) = \sum_{lm} \eta^k_{lm}(s) f^k_{lm}
\end{equation}

where $\eta^k_{lm}(s)$ is the interpolation factor from the solution grid
$(\phi^k_l,\zeta^k_m)$ to the path.

For some species, $k$, $\beta^k$ depends upon $f^k$.

For some species, we interpolate $f^k$ logarithmically from the grid to
the path, i.e.,

\begin{equation}
\alpha(s) = \sum_k \hat{f}^k(s)\, \beta^k(s) \text{ where }
 \hat{f}^k(s) =
 \exp\left( \sum_{lm} \eta^k_{lm}(s) \ln f^k_{lm} \right)
\end{equation}

\section{$\frac{\partial\alpha(s)}{\partial f^k_{lm}}$}

In Equation (\ref{100-dI_dfklm}) in \h{wvs-100}, the expression for the
derivative of the solution of the radiative-transfer equation with
respect to mixing ratio, the integral

\begin{equation}
\int_s^{s_m} \frac{\partial \alpha(\sigma)}{\partial f^k_{lm}}
 \d \sigma
\end{equation}

appears.  To incorporate TScat into the forward model, we will need
$\frac{\partial \omega_{0_{ij}}}{\partial f^k_{lm}}$ (see Equation
(\ref{095-twelve}) in \h{wvs-095}).  Since $\omega_{0_{ij}}$ depends upon
$\alpha(s)$, we need its derivatives for this purpose as well.

The combination of linear or logarithmic interpolation, and $\beta^k$
depending or not depending upon $f^k$, leads to four possibilities. 
Suppressing the superscript $k$ for clarity,

\begin{equation}
 \begin{array}{l|l}
 \alpha(s) & \frac{\partial \alpha(s)}{\partial f^k_{lm}}
\\[10pt]
 \hline
& \\[-5pt]
 \sum_k f(s)\, \beta(s) & 
 \beta(s) \eta_{lm}(s)
\\
 \sum_k f(s)\, \beta(s,f(s)) &
  \left( \beta(s,f(s)) + f(s) \frac{\partial \beta(s,f(s))}
                                         {\partial f(s)}
  \right) \eta_{lm}(s)
\\
 \sum_k \hat{f}(s)\, \beta(s) &
 \hat{f}(s)\, \beta(s)
    \frac{\eta_{lm}(s)}{f_{lm}}
\\
 \sum_k \hat{f}(s)\, \beta(s,\hat{f}(s)) &
 \hat{f}(s) \left( \beta(s,\hat{f}(s)) +
      \hat{f}(s)\frac{\partial \beta(s,\hat{f}(s))}
                       {\partial \hat{f}(s)} \right)
     \frac{\eta_{lm}(s)}{f_{lm}}
 \end{array}
\end{equation}

\section{$\frac{\partial^2\alpha(s)}
               {\partial f^k_{lm}\partial f^{\tilde k}_{\tilde l \tilde m}}$}

If $k \neq \tilde k$,
   $\frac{\partial^2\alpha(s)}
               {\partial f^k_{lm}\partial f^{\tilde k}_{\tilde l \tilde m}} =
               0$.

Suppressing the superscript $k$ and arguments of $\beta$ for clarity,

\begin{equation}
 \begin{array}{l|l}
 \alpha(s) & \frac{\partial^2\alpha(s)}
                  {\partial f^k_{lm}\partial f^k_{\tilde l \tilde m}}
\\[10pt]
 \hline
& \\[-5pt]
 \sum_k f(s)\, \beta(s) & 
 0
\\
 \sum_k f(s)\, \beta(s,f(s)) &
  \left( 2 \frac{\partial \beta}{\partial f(s)} +
         f(s) \frac{\partial^2 \beta}{\partial f(s)^2}
  \right) \eta_{lm}(s) \eta_{\tilde l \tilde m}(s)
\\
 \sum_k \hat{f}(s)\, \beta(s) &
 \beta \hat{f}(s)
  \frac{\eta_{lm}(s)}{f_{lm}}
  \frac{\eta_{\tilde l \tilde m}(s)}{f_{\tilde l \tilde m}} =
 \frac{\partial \alpha(s)}{\partial f_{lm}}
  \frac{\eta_{\tilde l \tilde m}(s)}{f_{\tilde l \tilde m}}
\\
 \sum_k \hat{f}(s)\, \beta(s,\hat{f}(s)) &
 \hat{f}(s) \left( \beta +
      3 \hat{f}(s)\frac{\partial \beta}
                       {\partial \hat{f}(s)} +
      (\hat{f}(s))^2 \frac{\partial^2 \beta}
                       {\partial \hat{f}(s)^2} \right)
     \frac{\eta_{lm}(s)}{f_{lm}}
     \frac{\eta_{\tilde l \tilde m}(s)}{f_{\tilde l \tilde m}} =
\\[7pt] &
 \left[
  \frac{\partial\alpha(s)}{\partial f_{lm}} +
  \hat{f}(s)^2 \left(
       2 \frac{\partial \beta}
             {\partial \hat{f}(s)} +
       \hat{f}(s) \frac{\partial^2 \beta}
                       {\partial \hat{f}(s)^2} \right)
     \frac{\eta_{lm}(s)}{f_{lm}}
 \right]
     \frac{\eta_{\tilde l \tilde m}(s)}{f_{\tilde l \tilde m}}
 \end{array}
\end{equation}

So far, we don't have any species for which
$\frac{\partial^2 \beta}{\partial \hat{f}(s)^2} \neq 0$.

\section{$\frac{\partial\T}{\partial f^k_{lm}}$}

\begin{equation}
\T(s_1,s_2) = \exp\left( -\int_{s_1}^{s_2} \alpha(s) \d s \right)\,.
\end{equation}

Suppressing the integral limits for clarity,

\begin{equation}
\frac{\partial\T}{\partial f^k_{lm}} =
 -\T \int \frac{\partial \alpha(s)}{\partial f^k_{lm}} \d s
\end{equation}

\section{$\frac{\partial^2\T}
               {\partial f^k_{lm}\partial f^{\tilde k}_{\tilde l \tilde m}}$}

\begin{equation}\begin{split}
\frac{\partial^2\T}{\partial f^k_{lm}\partial f^{\tilde k}_{\tilde l \tilde m}}
=\,&
\T \left( \int \frac{\partial \alpha(s)}{\partial f^k_{lm}} \d s
          \int \frac{\partial \alpha(s)}
                    {\partial f^{\tilde k}_{\tilde l \tilde m}} \d s -
          \int \frac{\partial^2 \alpha(s)}
                    {\partial f^k_{lm}
                     \partial f^{\tilde k}_{\tilde l \tilde m}} \d s
   \right) \\
=\,& \frac1{\T} \frac{\partial\T}{\partial f^k_{lm}}
                \frac{\partial\T}{\partial f^{\tilde k}_{\tilde l \tilde m}}
    -\T \int \frac{\partial^2 \alpha(s)}
                    {\partial f^k_{lm}
                     \partial f^{\tilde k}_{\tilde l \tilde m}} \d s
\end{split}\end{equation}
