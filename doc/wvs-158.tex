From the ATBD before Equation (5.11)

\begin{equation}
N(\phi) = \frac{a^2}{\sqrt{a^2 \cos^2 \phi + c^2 \sin^2 \phi}} \,.
\end{equation}

From ATBD Equation (5.10)

\begin{equation}
\vec{R}^\oplus =
 N(\phi) \left[ \cos \phi, \frac{c^2}{a^2} \sin \phi \right]
 = \left[ \frac{a^2 \cos \phi}
                   {\sqrt{a^2 \cos^2 \phi + c^2 \sin^2 \phi}},
              \frac{c^2 \sin \phi}
                   {\sqrt{a^2 \cos^2 \phi + c^2 \sin^2 \phi}} \right]
\end{equation}

Tangent vector

\begin{equation}
\vec{T} = \frac{\d \vec{R}^\oplus}{\d \phi} =
 \frac{a^2 c^2}{(a^2 \cos^2 \phi + c^2 \sin^2 \phi)^{\frac32}} \,
 [\, -\sin \phi, \cos \phi \, ] \,.
\end{equation}

Inward gradient vector

\begin{equation}
\vec{G} = [ \vec{T}_2, -\vec{T}_1 ] \,.
\end{equation}

$\vec{d}$ vector

\begin{equation}
\vec{d} = \vec{R}^\oplus - \vec{G} =
 N ( \phi ) \,
 \left[ \left(1 - \frac{c^2}{a^2 \cos^2 \phi + c^2 \sin^2 \phi} \right)
          \cos \phi,
        \left( \frac{c^2}{a^2}
          - \frac{c^2}{a^2 \cos^2 \phi + c^2 \sin^2 \phi} \right)
          \sin \phi
 \right] \,.
\end{equation}

From ATBD Equation (5.21)

\begin{equation}
R_\text{eq}^\oplus \equiv
H_t^\oplus = N(\phi_t) \sqrt{ \sin^2 \phi_t + \frac{c^4}{a^4} \cos^2 \phi_t}
 = \sqrt{ \frac{a^4 \sin^2 \phi_t + c^4 \cos^2 \phi_t}
               {a^2 \cos^2 \phi_t + c^2 \sin^2 \phi_t}}
\end{equation}

From ATBD Equation (5.22)

\begin{equation}\begin{split}
\vec{d} = \,& \left[ \left( N(\phi_t) - H_t^\oplus \right) \cos \phi_t,
                 \left( \frac{c^2}{a^2} 
                        N(\phi_t) - H_t^\oplus \right) \sin \phi_t \right]
        \\
 = \,& N(\phi_t) \left[ \left( 1 -
                    \sqrt{ \sin^2 \phi_t + \frac{c^4}{a^4} \cos^2 \phi_t}
                    \right) \cos \phi_t,
                    \left( \frac{c^2}{a^2} -
                    \sqrt{ \sin^2 \phi_t + \frac{c^4}{a^4} \cos^2 \phi_t}
                    \right) \sin \phi_t
             \right] \\
\end{split}\end{equation}
